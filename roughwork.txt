def filter_emp_page(request):
    if request.method == 'POST':
        nm=request.POST['nm']
        dp=request.POST['dp']
        ro=request.POST['ro']
        print(nm,dp,ro)
        emps=Employee.objects.all()
        if nm:
            emps=emps.filter(Q(first_name__icontains = nm) | Q(last_name__icontains = nm))
        if dp:
            emps=emps.filter(dept__name__icontains = dp)
        if ro:
            emps = emps.filter(role__name__icontains = ro)
        context={
            'emps': emps
        }
        return render(request,'allemp.html',context)

    elif request.method == 'GET':
        return render(request,'filteremp.html')
    else:
        return HttpResponse('An Exception Occured!!')
